	multiply = 4
	multiply = {

			limit = {
				upc_quota_predicted_percent >= 1
			}
			value = 1.2

		}
		else = {
			value = upc_quota_predicted_percent
			min = 0.05
		}

		limit = { has_variable = upc_anti_cheese }
		multiply = 1.2
	}
	multiply = 60
	
}
