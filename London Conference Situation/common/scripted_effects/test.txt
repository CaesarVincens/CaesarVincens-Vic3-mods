
# Change a sides victory condition text
# This effect has to be run inside a scope where 'scope:com_situation' is available!
# Parameters:
# - side = either 'left' or 'right'
# - name = localization key for the new victory condition text
set_com_situation_side_victory_condition_text = {
    hidden_effect = {
        if = {
            limit = {
                NOT = {
                    exists = scope:com_situation
                }
            }
            error_log = error_no_situation_scope
        }
        else = {
            fix_variable_error = { variable = $victory_condition$ }
            fix_variable_error = { variable = com_victory_condition }
            scope:com_situation = {
                var:com_side_$side$ = {
                    set_variable = {
                        name = com_victory_condition
                        value = flag:$victory_condition$
                    }
                }
            }
        }
    }
}
